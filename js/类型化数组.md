```js

// 申请12byte的内存
var buffer = new ArrayBuffer(12);

// 00000000, 00000000, 00000000, 00000000, 00000000, 00000000, 00000000, 00000000, 00000000, 00000000, 00000000, 00000000,

//              0,                             0                          0                      0

// 格式化为32bit, 即4个byte为数组的一个元素
// 也: 创建一个数组,其元素为32位的整型, 那么12字节的内存,可以是3个元素

var x = new Int32Array(buffer);
console.log(x); [0,0,0];

x[1] = 1234;
console.log(new Number(1234).toString(2));

// 00000000, 00000000, 00000000, 00000000, | 00000000, 00000000, 00000100, 11010010, | 00000000, 00000000, 00000000, 00000000,

var slice = buffer.slice(4);
var y = new Int32Array(slice);

// 00000000, 00000000, 00000100, 11010010, | 00000000, 00000000, 00000000, 00000000
console.log(y); [1134,0]

console.log(x[1]); 
console.log(y[0]);
x[1] = 6789;
console.log(x[1]); 
console.log(y[0]);
```
